#include <iostream>

using namespace std;

void main() {
	char arr[5][5] = { 
		"qwer",
		"asdf",
		"zxcv",
		"1234" };
	//здесь проходимся по строкам
	for (int i = 0; i < 5; i++) {
		cout << arr[i]<<endl;
	}

	//здесь проходимся по символам
	for (int i = 0; i < 5; i++) {
		for (int j = 0; j < 5; j++) {
			cout << arr[i][j]<<" ";
		}
		cout << endl;
	}



}
