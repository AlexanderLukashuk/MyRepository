#include <iostream>
using namespace std;
//5.*Пользователь вводит строку 
//символов и искомый символ, 
//посчитать сколько раз он встречается в строке.
void main() {
	char str[100];
	char x;
	int count=0;
	cout << "Enter string: " << endl;
	cin >> str;
	cout << "Enter char: " << endl;
	cin >> x;

	for (int i = 0; str[i] != '\0'; i++) {
		if (str[i] == x) {
			count++;
		}
	}

	cout << count << endl;
}
