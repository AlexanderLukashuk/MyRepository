#include <iostream>
using namespace std;

class TEST {
	int* mas;
	int size;
public:
	TEST() {
		size = 10;
		mas = new int[size];
		for (int i = 0; i < size; i++) {
			mas[i] = rand() % 100;
		}
	}
	~TEST() { delete[] mas; }

	int& operator[](int pos) {
		return mas[pos];
	}
	void show() {
		for (int i = 0; i < size; i++)
			cout << mas[i] << " ";
	}
};

int main() {
	TEST test;
	test.show();
	
	cout << endl << test[5] << endl;
	test[5] = 99999;

	test.show();
	cin.get();
}
