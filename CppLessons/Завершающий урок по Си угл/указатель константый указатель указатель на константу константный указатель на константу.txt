#include<iostream>
using namespace std;

int main() {
	int num;
	int num2;
	const int cnum = 10;

	int* ptr1 = &num;		// указатель
	*ptr1 = 10;				// запись переменной через указатель
	cout << *ptr1 << endl;  // чтение переменной через указатель
	ptr1 = &num2;			// перезапись указателя
	ptr1 = &cnum;			// не сохр. адресс конст. переменной


	const int* ptr2 = &num; // указатель на константу
	*ptr2 = 10;				// запись переменной через указатель
	cout << *ptr2 << endl;  // чтение переменной через указатель
	ptr2 = &num2;			// перезапись указателя
	ptr2 = &cnum;			// могу сохр. адресс конст. переменной


	int* const ptr3 = &num;	  // константый указатель
	*ptr3 = 10;				  // запись переменной через указатель
	cout << *ptr3 << endl;    // чтение переменной через указатель
	ptr3 = &num2;			  // перезапись указателя
	int* const ptr33 = &cnum; // не сохр. адресс конст. переменной
	

	const int* const ptr4 = &num;   // константый указатель на константу
	*ptr4 = 10;						// запись переменной через указатель
	cout << *ptr4 << endl;			// чтение переменной через указатель
	ptr4 = &num2;					// перезапись указателя
	const int* const ptr44 = &cnum;	// могу сохр. адресс конст. переменной


	cin.get();
	return 0;
}
